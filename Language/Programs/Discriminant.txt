MODULE Discriminant;

VARIABLES
  a: INT;
  b: INT;
  c: INT;
  discriminant: INT;

PROCEDURE CalculateDiscriminant(a1: INT, b1: INT, c1: INT) : INT
VARIABLES
  result : INT;
START
  result := b1 * b1 - 4 * a1 * c1;
  RETURN result;
END;

START
  discriminant := CalculateDiscriminant(a, b, c);
END Discriminant.
